# 迁移容器
## 在源机器上操作
### 1. 停止容器（可选但推荐）
```bash
docker stop xr
```
### 2. 将容器提交为新镜像
```bash
docker commit xr ros_humble_xr_agx:v1

docker commit xr ros_humble_xr_zme:v2           # arm64
docker commit xr ros_humble_xr_zme_x86:v1       # x64 amd64
```
### 3. 将镜像保存为 tar 文件
```bash
docker save ros_humble_xr_agx:v1 -o ros_humble_xr_agx.tar

docker save ros_humble_xr_zme:v2 -o ros_humble_xr_zme.tar
docker save ros_humble_xr_zme_x86:v1 -o ros_humble_xr_zme_x86.tar
```
### 4. 传输文件到目标机器
```bash
# 使用 scp 传输（示例）
scp ros_humble_xr_zme.tar zme@ip:/path/to/destination/
```
## 在目标机器上操作
### 5. 加载镜像
```bash
docker load -i ros_humble_xr_agx.tar

docker load -i ros_humble_xr_zme.tar
docker load -i ros_humble_xr_zme_x86.tar
```

### 6. 运行新容器
```bash
docker run -it -d --name xr \
  -w /projects \
  -v /home/zme/pico_tele_agx:/projects \
  -v /dev/bus/usb:/dev/bus/usb \
  --privileged \
  --net=host \
  ros_humble_xr_agx:v1

docker run -it -d --name xr \
  -w /projects \
  -v /home/zme/pico_tele_zme:/projects \
  -v /home/zme/robot_config/system_config/zme_robot/cyclonedds_config:/projects/cyclonedds_config \
  -v /dev/bus/usb:/dev/bus/usb \
  --privileged \
  --net=host \
  ros_humble_xr_zme:v2
  
docker run -it -d --name xr \
  -w /projects \
  -v /home/zme/pico_tele_zme:/projects \
  -v /home/zme/robot_config/system_config/zme_robot/cyclonedds_config:/projects/cyclonedds_config \
  -v /dev/bus/usb:/dev/bus/usb \
  --privileged \
  --net=host \
  ros_humble_xr_zme_x86:v1
  
  
source /opt/ros/humble/setup.bash
export RMW_IMPLEMENTATION=rmw_cyclonedds_cpp
export CYCLONEDDS_URI=file:///projects/cyclonedds_config/cyclonedds.xml
export ROS_DOMAIN_ID=29
```

### 7 安装docker
```shell
sudo apt update
sudo apt install -y docker.io

sudo groupadd docker               #添加用户组
sudo usermod -aG docker ${USER}    #将当前用户添加至用户组
newgrp docker                      #更新用户组
sudo service docker restart
```

### 8 松灵环境
```shell
sudo apt-get install terminator
sudo apt update && sudo apt install can-utils ethtool
sudo apt install -y kmod
sudo apt-get install -y iproute2
```

# 检测Pico连接情况

```shell
#!/bin/bash
# Pico USB 连接排查脚本 (Ubuntu) - 含自动安装 ADB 功能

echo "=========================================="
echo "        Pico USB 连接排查工具"
echo "=========================================="

# 检查 lsusb
if ! command -v lsusb &> /dev/null; then
    echo "❌ 未找到 lsusb 命令，请安装: sudo apt install usbutils"
    exit 1
fi

# 检查 adb，如果未安装则尝试自动安装
if ! command -v adb &> /dev/null; then
    echo "⚠️ 未找到 adb 命令（Android 调试桥）"
    read -p "是否现在安装 adb？(输入 y 确认，其他键跳过): " -n 1 -r
    echo   # 换行
    if [[ $REPLY =~ ^[Yy]$ ]]; then
        echo "🔄 正在安装 adb，请稍候..."
        sudo apt update && sudo apt install adb -y
        if command -v adb &> /dev/null; then
            echo "✅ adb 安装成功！"
        else
            echo "❌ adb 安装失败，请手动执行: sudo apt install adb"
            exit 1
        fi
    else
        echo "请手动安装 adb 后重新运行此脚本: sudo apt install adb"
        exit 1
    fi
fi

# 1. 检测 Pico 设备
echo "🔍 正在扫描 USB 设备中的 Pico ..."
PICO_LINE=$(lsusb | grep -i "PICO" | head -n 1)

if [ -z "$PICO_LINE" ]; then
    echo "❌ 未检测到任何 Pico 设备。"
    echo "   可能原因："
    echo "   - USB 线缆未连接或松动"
    echo "   - Pico 设备未开机"
    echo "   - 设备处于充电模式（需要文件传输模式）"
    echo "👉 请先检查硬件连接，确保设备已开机。"
    exit 0
fi

echo "✅ 检测到 Pico 设备："
echo "   $PICO_LINE"

# 2. 尝试获取 USB 配置状态
echo "🔍 正在通过 ADB 获取 USB 配置状态 ..."

RESULT=$(timeout 5 adb shell getprop persist.sys.usb.config 2>&1 || true)

if [[ "$RESULT" == *"error"* ]] || [[ "$RESULT" == *"not found"* ]] || [[ -z "$RESULT" ]]; then
    echo "⚠️ 无法获取 USB 配置，ADB 可能未连接或未授权。"
    echo "   请检查："
    echo "   - 是否已在 Pico 上开启“开发者选项”和“USB 调试”"
    echo "👉 当前卡在 ADB 连接/授权阶段。"
else
    CONFIG=$(echo "$RESULT" | tr -d '\r\n')
    echo "📋 当前 USB 配置: $CONFIG"

    if [[ "$CONFIG" == *"adb"* ]]; then
        echo "🎉 ADB 调试模式已开启，连接正常！"
        echo "   可以通过 Pico 遥操！"
    elif [[ "$CONFIG" == *"mtp"* ]]; then
        echo "⚠️ 当前为 MTP（文件传输）模式，ADB 未开启。"
        echo "   请检查："
        echo "   - Pico 系统版本需要5.14.4及以上，新机器需要更新系统”"
        echo "   - 是否已在 Pico 上开启“开发者选项”和“USB 调试”"
        echo "   - 检查 Pico 是否运行XRobotToolkit软件，且勾选“Shared network(connect USB first)”"
        echo "👉 当前卡在 ADB 未启用阶段。"
    else
        echo "⚠️ 未知的 USB 配置: $CONFIG"
        echo "   可能设备处于其他模式，请反馈开发人员。"
        echo "👉 当前卡在未知配置阶段。"
    fi
fi

echo "=========================================="
echo "排查结束。"
```
